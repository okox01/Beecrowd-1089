#include <stdio.h>
int main() {
    int n, num[100], count1 = 0;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &num[i]);
    }
    if (n == 2) {
        if (num[0] != num[1]) {
            printf("2\n");
        } else {
            printf("0\n");
        }
    } else {
        for (int i = 0; i < n; i++) {
            if ((num[i] > num[(i + 1)%n] && num[i] > num[(i + n - 1)%n]) || (num[i] < num[(i + 1)%n] && num[i] < num[(i + n - 1)%n])) {
                count1++;
            }
        }
        printf("%d\n", count1);
    }
    return 0;
}
